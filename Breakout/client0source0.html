        body { font-family: sans-serif; margin: 40px; }
        #log { width: 400px; height: 200px; border: 1px solid #ccc; overflow-y: scroll; padding: 10px; margin-top: 10px; background: #f9f9f9; }
    </style>
</head>
<body>

    <h2>WebSocket Client</h2>
    <input type="text" id="messageInput" placeholder="Type a message..." disabled>
    <button id="sendBtn" disabled>Send</button>
    
    <div id="log"></div>

    <script>
        // Connect to the Python WebSocket server URL
        //const socket = new WebSocket("ws://localhost:8765");
		const socket = new WebSocket("ws://localhost:8080/ws");
        const logDiv = document.getElementById("log");
        const input = document.getElementById("messageInput");
        const btn = document.getElementById("sendBtn");

        function log(text) {
            logDiv.innerHTML += `<div>${text}</div>`;
            logDiv.scrollTop = logDiv.scrollHeight;
        }

        // Connection opened event handler
        socket.onopen = (event) => {
            log("<b>Connected to server!</b>");
            input.disabled = false;
            btn.disabled = false;
        };

        // Listen for messages from the server
        socket.onmessage = (event) => {
            log(`<span style="color: blue;">⬇️ ${event.data}</span>`);
        };

        // Connection closed event handler
        socket.onclose = (event) => {
            log("<b style='color: red;'>Disconnected from server.</b>");
            input.disabled = true;
            btn.disabled = true;
        };

        // Send message to the server
        function sendMessage() {
            const message = input.value;
            if (message) {
                socket.send(message);
                log(`<span style="color: green;">⬆️ Sent: ${message}</span>`);
                input.value = "";
            }
        }

        btn.addEventListener("click", sendMessage);
        input.addEventListener("keypress", (e) => { if (e.key === "Enter") sendMessage(); });
    </script>
</body>
